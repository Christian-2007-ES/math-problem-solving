	\section{Problema 1}
	Sean $a,b,c$ números reales positivos. Probar la desigualdad
	\begin{center}
		$\dfrac{1}{a(b+1)}+\dfrac{1}{b(c+1)}+\dfrac{1}{c(a+1)}\geq\dfrac{3}{1+abc}.$
	\end{center}
	
	\textbf{Solución:} Factorizando:
	\[
	(x+1)(x+2)=0
	\]
	Por tanto,
	\[
	x=-1 \quad \text{o} \quad x=-2
	\]
